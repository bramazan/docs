## Expected Usage

### Installation

```bash
npm install @odeal/payment-sdk
# or
yarn add @odeal/payment-sdk
```

### Basic Usage

```javascript
import { Odeal } from '@odeal/payment-sdk';

const odeal = new Odeal({
  apiKey: process.env.ODEAL_API_KEY,
  environment: 'production' // or 'test'
});

// Create a payment
const payment = await odeal.payments.create({
  merchantId: 'merchant-123',
  amount: 100.00,
  currency: 'TRY',
  orderId: 'order-456'
});

// Process with card
const result = await odeal.payments.process(payment.id, {
  card: {
    number: '4111111111111111',
    holderName: 'John Doe',
    expiryMonth: '12',
    expiryYear: '2025',
    cvv: '123'
  }
});

console.log(result.status); // 'SUCCEEDED'
```

##Current Alternative: REST API

Until the SDK is released, use our REST API:

```javascript
const fetch = require('node-fetch');

async function createPayment(data) {
  const response = await fetch('https://api.odeal.com/v1/payments', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.ODEAL_API_KEY}`,
      'Content-Type': 'application/json',
      'X-Idempotency-Key': `order-${data.orderId}-${Date.now()}`
    },
    body: JSON.stringify(data)
  });
  
  return await response.json();
}
```
